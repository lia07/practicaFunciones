numeroPal = int(input("Buscar palabras en una lista de cadenas:"))
if numeroPal < 1:
    print("¡NUL!")
else:
    lista = []
    for i in range(numeroPal):
        print("La palabra es aladd", str(i+1)+":", end = "")
        palabra = input()
        lista+= [palabra]
    print("La lista creada es:", lista)
